module HestonCook where

import Distributions.bug
tmax = 10.0
dt = 0.01

samples= 2000

heston = do
   k        ~ gamma 1 1
   th       ~ gamma 1 0.1
   eta      ~ gamma 1 0.1
   mu       ~ normal 0 0.1
   w1       ~  wiener
   w2       ~  wiener
   v_0      ~  gamma 1 0.1
   s_0      ~  uniform 0.0 2.0
   D v t = k * (th - v t) + eta * sqrt (v t) * d w1 t
   D s t = mu * s t + sqrt (v t) * s t * d w2 t
   return s
