import Distributions.bug

heston = do
   k        ~ gamma 1 1
   th       ~ gamma 1 0.1
   eta      ~ gamma 1 0.1
   mu       ~ normal 0 0.1
   w1       ~  wiener
   w2       ~  wiener
   v_0      ~  gamma 1 0.1
   s_0      ~  uniform 0.0 2.0
   D v t = k * (th - v t) + eta * sqrt (v t) * d w1 t
